class RabinKarp {

    public static void search(String text, String pattern) {
        int n = text.length();
        int m = pattern.length();

        int base = 256;   // number of characters
        int mod = 101;    // prime number

        int patternHash = 0;
        int textHash = 0;
        int h = 1;

        // Calculate h = base^(m-1) % mod
        for (int i = 0; i < m - 1; i++) {
            h = (h * base) % mod;
        }

        // Calculate initial hash values
        for (int i = 0; i < m; i++) {
            patternHash = (base * patternHash + pattern.charAt(i)) % mod;
            textHash = (base * textHash + text.charAt(i)) % mod;
        }

        // Slide the pattern over text
        for (int i = 0; i <= n - m; i++) {

            // If hash matches, check characters
            if (patternHash == textHash) {
                int j;
                for (j = 0; j < m; j++) {
                    if (text.charAt(i + j) != pattern.charAt(j)) {
                        break;
                    }
                }

                if (j == m) {
                    System.out.println("Pattern found at index: " + i);
                }
            }

            // Calculate next window hash
            if (i < n - m) {
                textHash = (base * (textHash - text.charAt(i) * h)
                        + text.charAt(i + m)) % mod;

                if (textHash < 0) {
                    textHash += mod;
                }
            }
        }
    }

    public static void main(String[] args) {
        String text = "ABCCDDAEFG";
        String pattern = "CDD";

        search(text, pattern);
    }
}
